# Mira MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Mira MVP — staff record push-to-talk voice reports that deliver instantly to team channels, then get transcribed and AI-classified in the background, with a command center showing the live operations feed.

**Architecture:** Next.js 15 (App Router) on Vercel; browser MediaRecorder captures audio, uploads to S3 via a Next.js API route, message is immediately written to DynamoDB and visible in the channel feed; a second fire-and-forget API call handles Groq Whisper transcription + Claude Haiku enrichment asynchronously; channel feeds poll DynamoDB every 2 seconds.

**Tech Stack:** Next.js 15, TypeScript, Tailwind CSS, AWS DynamoDB, AWS S3, Groq (Whisper), Anthropic Claude Haiku (`claude-haiku-4-5-20251001`), Vitest

- [ ] **Step 3: Create src/lib/dynamo.ts**

  ```typescript
  import { DynamoDBClient } from '@aws-sdk/client-dynamodb'
  import {
    DynamoDBDocumentClient,
    PutCommand,
    GetCommand,
    UpdateCommand,
    QueryCommand,
    ScanCommand,
  } from '@aws-sdk/lib-dynamodb'
  import type { Channel, Team, Message } from './types'

  const client = new DynamoDBClient({
    region: process.env.AWS_REGION!,
    credentials: {
      accessKeyId: process.env.AWS_ACCESS_KEY_ID!,
      secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY!,
    },
  })

  const db = DynamoDBDocumentClient.from(client)

  const TABLES = {
    channels: 'MiraChannels',
    teams: 'MiraTeams',
    messages: 'MiraMessages',
  } as const

  export function buildUpdateExpression(updates: Record<string, unknown>): {
    UpdateExpression: string
    ExpressionAttributeNames: Record<string, string>
    ExpressionAttributeValues: Record<string, unknown>
  } {
    const entries = Object.entries(updates).filter(([, v]) => v !== undefined)
    return {
      UpdateExpression: `SET ${entries.map(([k]) => `#${k} = :${k}`).join(', ')}`,
      ExpressionAttributeNames: Object.fromEntries(entries.map(([k]) => [`#${k}`, k])),
      ExpressionAttributeValues: Object.fromEntries(entries.map(([k, v]) => [`:${k}`, v])),
    }
  }

  export async function putTeam(team: Team): Promise<void> {
    await db.send(new PutCommand({ TableName: TABLES.teams, Item: team }))
  }

  export async function putChannel(channel: Channel): Promise<void> {
    await db.send(new PutCommand({ TableName: TABLES.channels, Item: channel }))
  }

  export async function listChannels(): Promise<Channel[]> {
    const result = await db.send(new ScanCommand({ TableName: TABLES.channels }))
    return (result.Items ?? []) as Channel[]
  }

  export async function putMessage(message: Message): Promise<void> {
    await db.send(new PutCommand({ TableName: TABLES.messages, Item: message }))
  }

  export async function getMessage(messageId: string): Promise<Message | null> {
    const result = await db.send(
      new GetCommand({ TableName: TABLES.messages, Key: { messageId } })
    )
    return (result.Item as Message) ?? null
  }

  export async function getChannelMessages(channelId: string, limit = 50): Promise<Message[]> {
    const result = await db.send(
      new QueryCommand({
        TableName: TABLES.messages,
        IndexName: 'channelId-createdAt-index',
        KeyConditionExpression: 'channelId = :cid',
        ExpressionAttributeValues: { ':cid': channelId },
        ScanIndexForward: false,
        Limit: limit,
      })
    )
    return ((result.Items ?? []) as Message[]).reverse()
  }

  export async function updateMessageEnrichment(
    messageId: string,
    fields: Partial<Pick<Message, 'transcript' | 'category' | 'urgency' | 'location' | 'summary' | 'suggestedOwner' | 'confidence' | 'status'>>
  ): Promise<void> {
    const { UpdateExpression, ExpressionAttributeNames, ExpressionAttributeValues } =
      buildUpdateExpression(fields as Record<string, unknown>)

    await db.send(
      new UpdateCommand({
        TableName: TABLES.messages,
        Key: { messageId },
        UpdateExpression,
        ExpressionAttributeNames,
        ExpressionAttributeValues,
      })
    )
  }
  ```

  ```typescript
  import { S3Client, PutObjectCommand, GetObjectCommand } from '@aws-sdk/client-s3'
  import { getSignedUrl } from '@aws-sdk/s3-request-presigner'

  const s3 = new S3Client({
    region: process.env.AWS_REGION!,
    credentials: {
      accessKeyId: process.env.AWS_ACCESS_KEY_ID!,
      secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY!,
    },
  })

  const BUCKET = process.env.AWS_S3_BUCKET!

  export async function uploadAudio(key: string, buffer: Buffer, contentType: string): Promise<void> {
    await s3.send(
      new PutObjectCommand({ Bucket: BUCKET, Key: key, Body: buffer, ContentType: contentType })
    )
  }

  export async function downloadAudio(key: string): Promise<Buffer> {
    const result = await s3.send(new GetObjectCommand({ Bucket: BUCKET, Key: key }))
    const chunks: Buffer[] = []
    for await (const chunk of result.Body as AsyncIterable<Uint8Array>) {
      chunks.push(Buffer.from(chunk))
    }
    return Buffer.concat(chunks)
  }

  export async function getAudioUrl(key: string): Promise<string> {
    return getSignedUrl(
      s3,
      new GetObjectCommand({ Bucket: BUCKET, Key: key }),
      { expiresIn: 3600 }
    )
  }
  ```

  ```typescript
  import type { UrgencyLevel } from './types'

  const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY! })

  export interface EnrichmentResult {
    category: string
    urgency: UrgencyLevel
    location: string
    summary: string
    suggestedOwner: string
    confidence: number
  }

  export function parseEnrichmentResponse(text: string, fallbackSummary: string): EnrichmentResult {
    try {
      return JSON.parse(text) as EnrichmentResult
    } catch {
      return {
        category: 'general',
        urgency: 'low',
        location: '',
        summary: fallbackSummary.slice(0, 100),
        suggestedOwner: '',
        confidence: 0.5,
      }
    }
  }

  export async function enrichTranscript(
    transcript: string,
    channelName: string
  ): Promise<EnrichmentResult> {
    const message = await anthropic.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 256,
      messages: [
        {
          role: 'user',
          content: `You are an operations intelligence system for live events. Extract structured information from this voice report.

  Channel: ${channelName}
  Transcript: "${transcript}"

  Respond with ONLY a JSON object, no markdown:
  {
    "category": "one of: security, medical, cleanup, logistics, av, maintenance, food_beverage, general",
    "urgency": "one of: low, medium, high, critical",
    "location": "location mentioned, or empty string if none",
    "summary": "one sentence summary of the issue",
    "suggestedOwner": "team name best suited to handle this",
    "confidence": 0.0 to 1.0
  }`,
        },
      ],
    })

    const text = message.content[0].type === 'text' ? message.content[0].text : ''
    return parseEnrichmentResponse(text, transcript)
  }
  ```

- [ ] **Step 1: Create src/app/api/reports/route.ts**

  ```typescript
  import { NextRequest, NextResponse } from 'next/server'
  import { randomUUID } from 'crypto'
  import { uploadAudio } from '@/lib/s3'
  import { putMessage } from '@/lib/dynamo'
  import type { Message } from '@/lib/types'

  export async function POST(request: NextRequest) {
    const formData = await request.formData()
    const audio = formData.get('audio') as File | null
    const channelId = formData.get('channelId') as string | null
    const senderId = formData.get('senderId') as string | null
    const senderName = formData.get('senderName') as string | null

    if (!audio || !channelId || !senderId || !senderName) {
      return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
    }

    const messageId = randomUUID()
    const audioKey = `audio/${messageId}.webm`
    const now = new Date().toISOString()

    const buffer = Buffer.from(await audio.arrayBuffer())
    await uploadAudio(audioKey, buffer, audio.type || 'audio/webm')

    const message: Message = {
      messageId,
      channelId,
      senderId,
      senderName,
      audioKey,
      deliveredAt: now,
      createdAt: now,
      status: 'delivered',
    }

    await putMessage(message)

    return NextResponse.json({ messageId, channelId, deliveredAt: now }, { status: 201 })
  }
  ```

## Task 6: Enrichment API

Background endpoint: downloads audio from S3, transcribes with Groq, enriches with Claude, updates DynamoDB. Called fire-and-forget from the client.

**Files:**
- Create: `src/app/api/reports/[id]/enrich/route.ts`

- [ ] **Step 1: Create src/app/api/reports/[id]/enrich/route.ts**

  ```typescript
  import { NextRequest, NextResponse } from 'next/server'
  import { getMessage, updateMessageEnrichment } from '@/lib/dynamo'
  import { downloadAudio } from '@/lib/s3'
  import { transcribeAudio } from '@/lib/transcribe'
  import { enrichTranscript } from '@/lib/enrich'

  export async function POST(
    _request: NextRequest,
    { params }: { params: Promise<{ id: string }> }
  ) {
    const { id } = await params

    const message = await getMessage(id)
    if (!message) {
      return NextResponse.json({ error: 'Message not found' }, { status: 404 })
    }
    if (message.status !== 'delivered') {
      return NextResponse.json({ skipped: true }, { status: 200 })
    }

    const audioBuffer = await downloadAudio(message.audioKey)

    const transcript = await transcribeAudio(audioBuffer, 'audio/webm')
    await updateMessageEnrichment(id, { transcript, status: 'transcribed' })

    const enrichment = await enrichTranscript(transcript, message.channelId)
    await updateMessageEnrichment(id, { ...enrichment, status: 'enriched' })

    return NextResponse.json({ success: true })
  }
  ```

  > **Note:** Next.js 15 requires `params` to be awaited in route handlers. Always use `const { id } = await params`.

  ```tsx
  'use client'

  import { useState, useRef, useCallback } from 'react'

  interface PTTButtonProps {
    channelId: string
    senderId: string
    senderName: string
    onReportCreated?: (messageId: string) => void
  }

  export function PTTButton({ channelId, senderId, senderName, onReportCreated }: PTTButtonProps) {
    const [isRecording, setIsRecording] = useState(false)
    const [isUploading, setIsUploading] = useState(false)
    const mediaRecorder = useRef<MediaRecorder | null>(null)
    const chunks = useRef<Blob[]>([])

    const startRecording = useCallback(async () => {
      try {
        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
        const recorder = new MediaRecorder(stream)
        chunks.current = []
        recorder.ondataavailable = (e) => {
          if (e.data.size > 0) chunks.current.push(e.data)
        }
        recorder.start(100)
        mediaRecorder.current = recorder
        setIsRecording(true)
      } catch {
        alert('Microphone access required')
      }
    }, [])

    const stopRecording = useCallback(async () => {
      if (!mediaRecorder.current || !isRecording) return
      setIsRecording(false)
      setIsUploading(true)

      await new Promise<void>((resolve) => {
        mediaRecorder.current!.onstop = () => resolve()
        mediaRecorder.current!.stop()
        mediaRecorder.current!.stream.getTracks().forEach((t) => t.stop())
      })

      const blob = new Blob(chunks.current, { type: 'audio/webm' })
      const formData = new FormData()
      formData.append('audio', blob, 'report.webm')
      formData.append('channelId', channelId)
      formData.append('senderId', senderId)
      formData.append('senderName', senderName)

      try {
        const res = await fetch('/api/reports', { method: 'POST', body: formData })
        const { messageId } = await res.json()

        // Fire enrichment in background — no await, we don't block on it
        fetch(`/api/reports/${messageId}/enrich`, { method: 'POST' }).catch(() => {})

        onReportCreated?.(messageId)
      } finally {
        setIsUploading(false)
      }
    }, [channelId, senderId, senderName, isRecording, onReportCreated])

    let label = '🎙 Hold to Talk'
    let bgClass = 'bg-blue-600 hover:bg-blue-700 active:scale-95'
    if (isRecording) {
      label = '🔴 Recording...'
      bgClass = 'bg-red-600 scale-110 shadow-xl shadow-red-500/40'
    } else if (isUploading) {
      label = '⏳ Sending...'
      bgClass = 'bg-gray-600 cursor-not-allowed'
    }

    return (
      <button
        onMouseDown={startRecording}
        onMouseUp={stopRecording}
        onTouchStart={(e) => { e.preventDefault(); startRecording() }}
        onTouchEnd={(e) => { e.preventDefault(); stopRecording() }}
        disabled={isUploading}
        className={`select-none rounded-full w-28 h-28 text-white font-bold text-sm transition-all duration-100 ${bgClass}`}
      >
        {label}
      </button>
    )
  }
  ```

- [ ] **Step 3: Create src/components/ChannelFeed.tsx**

  ```tsx
  'use client'

  import { useEffect, useState, useCallback } from 'react'
  import { MessageCard } from './MessageCard'
  import type { Message } from '@/lib/types'

  interface ChannelFeedProps {
    channelId: string
  }

  export function ChannelFeed({ channelId }: ChannelFeedProps) {
    const [messages, setMessages] = useState<Message[]>([])
    const [error, setError] = useState(false)

    const fetchMessages = useCallback(async () => {
      try {
        const res = await fetch(`/api/channels/${channelId}/messages`)
        if (!res.ok) { setError(true); return }
        const data = await res.json()
        setMessages(data.messages)
        setError(false)
      } catch {
        setError(true)
      }
    }, [channelId])

    useEffect(() => {
      fetchMessages()
      const interval = setInterval(fetchMessages, 2000)
      return () => clearInterval(interval)
    }, [fetchMessages])

    if (error) return <p className="text-xs text-red-400 p-2">Connection error</p>
    if (messages.length === 0) {
      return <p className="text-xs text-gray-600 p-2 italic">No reports yet</p>
    }

    return (
      <div className="flex flex-col">
        {messages.map((msg) => (
          <MessageCard key={msg.messageId} message={msg} />
        ))}
      </div>
    )
  }
  ```
